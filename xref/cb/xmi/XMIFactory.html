171 	protected void setVisibility(AccessQualified a, MModelElement e) {
172 		String acc = a.getExportControl();
173 
174 		if (acc == null)
175 			acc = "public";
176 		else
177 			acc = acc.toLowerCase();
178 
179 		MVisibilityKind kind = MVisibilityKind.forName(acc);
180 
181 		if (kind == null)
182 			throw new RuntimeException("Can't map access qualifier: " + acc);
183 
184 		e.setVisibility(kind);
185 	}

204 	 * @return MInterface, MClass, MAssociationClass or MActor
205 	 */
206 	public MClassifier createClass(cb.petal.Class clazz) {
207 		MClassifier cl;
208 
209 		/*
210 		 * ArgoUML/Poseidon can't display/use Actors/AssociationClass correctly
211 		 */
212 		if (clazz.isInterface())
213 			cl = new MInterfaceImpl();
214 		else if (clazz.isActor())
215 			cl = new MActorImpl();
216 		else if (clazz.isAssociationClass())
217 			cl = new MAssociationClassImpl();
218 		else
219 			cl = new MClassImpl();
220 
221 		cl.setName(clazz.getNameParameter());
222 		cl.setUUID(clazz.getQuid());
223 
224 		setDocumentation(clazz, cl);
225 		setVisibility(clazz, cl);
226 		if (clazz.getPropertyAsBoolean("abstract"))
227 			cl.setAbstract(true);
228 		if (clazz instanceof ClassUtility)
229 			cl.setStereotype(getStereotype("utility"));
230 		else if (!(clazz.isInterface() || clazz.isActor()))
231 			setStereotype(clazz, cl);
232 
233 		return cl;
234 	}

306 		a.setUUID(attr.getQuid());
307 		setVisibility(attr, a);
308 		setStereotype(attr, a);
309 		setDocumentation(attr, a);
310 		a.setOwnerScope(attr.getStatic() ? MScopeKind.CLASSIFIER
311 				: MScopeKind.INSTANCE);
312 
313 		if (attr.getType() != null) {
314 			String type = attr.getType();
315 
316 			a.setType(getClassifierFor(type));
317 		}
318 
319 		String init = attr.getInitialValue();
320 
321 		if (init != null)
322 			a.setInitialValue(new MExpression("Java", init));
323 
324 		if (attr.getPropertyAsBoolean("derived") == true) {
325 			addTaggedValue(a, "derived", "true");
326 		}
327 
328 		return a;

338 	public MOperation createOperation(Operation op) {
339 		MOperation m = new MOperationImpl();
340 
341 		m.setName(op.getNameParameter());
342 		m.setUUID(op.getQuid());
343 
344 		setVisibility(op, m);
345 		setStereotype(op, m);
346 		setConcurrency(op, m);
347 		setDocumentation(op, m);
348 
349 		String type = (op.getResult() == null) ? "void" : op.getResult();
350 
351 		MParameter ret = new MParameterImpl();
352 		ret.setName("return");
353 		ret.setType(getClassifierFor(type));
354 		ret.setKind(MParameterDirectionKind.RETURN);
355 		m.addParameter(ret);
356 
357 		ArrayList params = new ArrayList();
358 
359 		if (op.getParameters() != null) {
360 			for (Iterator i = op.getParameters().getElements().iterator(); i
361 					.hasNext();) {
362 				cb.petal.Parameter p = (cb.petal.Parameter) i.next();
363 
364 				String name = p.getNameParameter();
365 				type = p.getType();
366 
367 				// Typical erroneous "Address a", needs to be split
368 
369 				if (type == null) {
370 					int index = name.indexOf(' ');
371 
372 					if (index > 0) {
373 						type = name.substring(0, index);
374 						name = name.substring(index + 1).trim();
375 					} else
376 						type = "int";
377 				}
378 
379 				MParameter param = new MParameterImpl();
380 				param.setName(name);
381 				param.setType(getClassifierFor(type));
382 				param.setKind(MParameterDirectionKind.IN);
383 				m.addParameter(param);
384 
385 			}
386 		}
387 		// Create Exceptions
388 		if (op.getExceptions() != null) {
389 			for (Iterator ie = op.getExceptions().getElements().iterator(); ie
390 					.hasNext();) {
391 				StringLiteral expName = (StringLiteral) ie.next();
392 				Pattern pattern = Pattern.compile("([^\\[]+)\\[(\\w+)\\]");
393 				Matcher matcher = pattern.matcher(expName.getValue());
394 				if (matcher.matches()) {
395 					String exp = matcher.group(1);
396 					MClassifier expClass = (MClassifier) getClassifierFor(exp);
397 					MSignal signal = new MSignalImpl();
398 					signal.setNamespace(expClass.getNamespace());
399 					signal.setName(expClass.getName());
400 					m.addRaisedSignal(signal);
401 				}
402 			}
403 		}
404 
405 		return m;
406 	}

408 	private void setupRelationship(MDependency g, Relationship rel) {
409 		MModelElement client = gen.getContainingClassifier(rel);
410 		MClassifier supplier = gen.getClassifier(rel.getQuidu());
411 
412 		// if client not found as class try as package
413 		if (client == null)
414 			client = gen.getContainingPackage(rel);
415 		g.addSupplier(supplier);
416 		g.addClient(client);
417 		g.setUUID(rel.getQuid());
418 
419 		setStereotype(rel, g);
420 
421 		if (rel.getLabel() != null)
422 			g.setName(rel.getLabel());
423 	}

457 		if ("n".equals(number.toLowerCase()) || "*".equals(number))
458 			return "" + MMultiplicity.N;
459 		else
460 			return number;
461 	}

463 	protected MMultiplicity getMultiplicityFor(Role role) {
464 		int from = 1, to = 1;
465 
466 		if (role.getClientCardinality() != null) {
467 			String card = role.getClientCardinality().getStringValue();
468 			StringTokenizer tok = new StringTokenizer(card, ".,");
469 
470 			try {
471 				from = Integer.parseInt(map(tok.nextToken()));
472 
473 				if (tok.hasMoreTokens())
474 					to = Integer.parseInt(map(tok.nextToken()));
475 				else
476 					to = from;
477 			} catch (Exception e) {
478 				throw new RuntimeException("Invalid cardinality: " + card);
479 			}
480 		} else {
481 			Role other = role.getOtherRole();
482 
483 			if (other.isAggregate()) {
484 				from = 0;
485 				to = MMultiplicity.N;
486 			}
487 		}
488 
489 		if (from == MMultiplicity.N)
490 			from = 0;
491 
492 		return new MMultiplicity(from, to);
493 	}

507 	public MAssociationEnd createAssociationEnd(Role role) {
508 		MAssociationEnd ae = new MAssociationEndImpl();
509 		ae.setNavigable(role.isNavigable());
510 		ae.setMultiplicity(getMultiplicityFor(role));
511 		if (!role.getRoleName().startsWith("$")) // Anonymous name
512 			ae.setName(role.getRoleName());
513 		if (role.isAggregation())
514 			ae.setAggregation(MAggregationKind.AGGREGATE);
515 		else if (role.isComposition())
516 			ae.setAggregation(MAggregationKind.COMPOSITE);
517 		else
518 			ae.setAggregation(MAggregationKind.NONE);
519 		String c = role.getConstraints();
520 		if (c != null) {
521 			MConstraint constr = new MConstraintImpl();
522 			MBooleanExpression body = new MBooleanExpression("Java", c);
523 			constr.setBody(body);
524 			ae.addConstraint(constr);
525 			gen.getPackage().addOwnedElement(constr);
526 		}
527 
528 		PetalNode qualifierNode = role.getProperty("keys");
529 		if (qualifierNode != null) {
530 			List qualifierList = (List) qualifierNode;
531 			for (Iterator iq = qualifierList.getElements().iterator(); iq.hasNext();) {
532 				ClassAttribute attribute = (ClassAttribute) iq.next();
533 
534 				MAttribute attr = createAttribute(attribute);
535 				ae.addQualifier(attr);
536 			}
537 		}
538 		return ae;
539 	}

568 	/**
569 	 * Convert Rose identifier to normal one.
570 	 */
571 	protected String makeName(String name) {
572 		char[] chars = name.toCharArray();
573 		StringBuffer buf = new StringBuffer();
574 
575 		for (int i = 0; i < chars.length; i++) {
576 			char ch = chars[i];
577 
578 			if ((ch == ':') && (chars[i + 1] == ':')) {
579 				buf.append('.');
580 				i++;
581 			} else {
582 				if (Character.isLetterOrDigit(ch))
583 					buf.append(ch);
584 				else
585 					buf.append('_');
586 			}
587 		}
588 
589 		return buf.toString();
590 	}

592 	/**
593 	 * Convert fully qualified rose name (with "foo::bar")
594 	 * 
595 	 * @return tuple (class name, package name)
596 	 */
597 	protected String[] makeNames(String qual) {
598 		String name, pack;
599 		int index = qual.indexOf("::");
600 
601 		if (index < 0) { // weird ...
602 			pack = "";
603 			name = makeName(qual);
604 		} else {
605 			int index2 = qual.lastIndexOf("::");
606 
607 			if (index == index2)
608 				pack = "";
609 			else
610 				pack = makeName(qual.substring(index + 2, index2));
611 
612 			name = qual.substring(index2 + 2);
613 		}
614 
615 		return new String[] { name, pack };
616 	}
