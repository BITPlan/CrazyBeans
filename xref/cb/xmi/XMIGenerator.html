52  /**
53   * Convert a Rose petal file into the <a
54   * href="http://xml.coverpages.org/xmi.html">XMI</a> format using NovoSoft's <a
55   * href="http://nsuml.sourceforge.net/"> NSUML</a> package which implements the
56   * entities defined in the <a href="http://www.omg.org/uml/">UML</a>
57   * specification 1.3.
58   *
59   * @version $Id: XMIGenerator.java,v 1.8 2005/05/09 20:34:55 moroff Exp $
60   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
61   */
62  public class XMIGenerator extends GeneratorVisitor implements Generator {

92  	protected Map<QuidObject,MPackage> package_map = new HashMap<QuidObject,MPackage>(); // Map<ClassCategory, MPackage>

94  	/**

96  	 * @param quid

100 		quid_map.put(quid, obj);
101 	}

104 	 * remove an Object
105 	 * @param quid
106 	 */
107 	protected void removeObject(String quid) {
108 		quid_map.remove(quid);
109 	}

112 	 * lookup a classifier by quid
113 	 * @param quid
114 	 * @return - the classifier found 
115 	 */
116 	protected final MClassifier getClassifier(String quid) {
117 		return quid_map.get(quid);
118 	}

120 	/**
121 	 * get the Package for the given quid
122 	 * @param quid - the quid to lookup
123 	 * @return - the package
124 	 */
125 	@SuppressWarnings("rawtypes")
126 	protected final MPackage getPackage(String quid) {
127 		// FIXME performance and readbility could be improved with another Map
128 		for (Iterator iter = package_map.values().iterator(); iter.hasNext();) {
129 			java.lang.Object element = (java.lang.Object) iter.next();
130 
131 			if (element instanceof MPackage) {
132 				MPackage modelPackage = (MPackage) element;
133 
134 				if (modelPackage.getUUID().equals(quid))
135 					return modelPackage;
136 			}
137 		}
138 
139 		return null;
140 	}

155 		factory = getFactory();

157 		pack = model;

160 	/**
161 	 * Start generation of XMI code.
162 	 */
163 	@Override
164 	public void start() {
165 		/*
166 		 * Run a first pass visitor to add all packages, classes, use cases, etc.
167 		 * that may be referenced from different places, or even be referenced in a
168 		 * forward declaration.
169 		 */
170 		getTree().accept(new DescendingVisitor() {
171 			private void addPackage(QuidObject obj, MPackage p) {
172 				package_map.put(obj, p);
173 
174 				pack.addOwnedElement(p);
175 				packages.push(pack); // Save old value
176 				pack = p;
177 
178 				super.visitObject(obj); // Default traversal
179 
180 				pack = (MPackage) packages.pop();
181 			}
182 
183 			public void visit(LogicalCategory obj) {
184 				addPackage(obj, factory.createPackage(obj));
185 			}
186 
187 			public void visit(UseCaseCategory obj) {
188 				addPackage(obj, factory.createPackage(obj));
189 			}
190 
191 			public void visit(SubSystem sub) {
192 				/**
193 				 * ArgoUML can't handle subsystems
194 				 */
195 				addPackage(sub, factory.createPackage(sub));
196 			}
197 
198 			public void visit(cb.petal.Class clazz) {
199 				String quid = clazz.getQuid();
200 
201 				MClassifier cl = factory.createClass(clazz);
202 				pack.addOwnedElement(cl);
203 				addObject(quid, cl);
204 			}
205 
206 			public void visit(UseCase caze) {
207 				String quid = caze.getQuid();
208 
209 				MClassifier cl = factory.createUseCase(caze);
210 				pack.addOwnedElement(cl);
211 				addObject(quid, cl);
212 			}
213 
214 			public void visit(Module module) {
215 				String quid = module.getQuid();
216 
217 				MClassifier cl = factory.createComponent(module);
218 				pack.addOwnedElement(cl);
219 				addObject(quid, cl);
220 			}
221 		});
222 
223 		getTree().accept(this);
224 	}

233 	private void setPackage(QuidObject obj) {
234 		MPackage p = (MPackage) package_map.get(obj);
235 
236 		packages.push(pack); // Save old value
237 		pack = p;
238 
239 		visitObject(obj); // Default traversal
240 
241 		pack = (MPackage) packages.pop();
242 	}

292 	/**
293 	 * If this association contains an association class, use that object,
294 	 * otherwise create new object.
295 	 */
296 	public void visit(Association assoc) {
297 		MAssociation a;
298 
299 		/*
300 		 * ArgoUML/Poseidon can't handle AssociationClass correctly
301 		 */
302 		cb.petal.Class clazz = assoc.getAssociationClass();
303 
304 		if (clazz != null) {
305 			a = (MAssociation) getClassifier(clazz.getQuid());
306 			factory.setupAssociation(assoc, a);
307 		} else {
308 			a = factory.createAssociation(assoc);
309 			pack.addOwnedElement(a);
310 		}
311 	}

315 	/**
316 	 * Search for element of given name in model (and all sub-packages)
317 	 * 
318 	 * @param name
319 	 *          name to look for with getName()
320 	 * @param clazz
321 	 *          Class searched element is an instance of
322 	 * @return found element or null
323 	 */
324 	public MModelElement searchElement(String name, java.lang.Class<?> clazz) {
325 		return searchElement(model, name, clazz);
326 	}
327 
328 	private static MModelElement searchElement(MPackage pack, String name,
329 			java.lang.Class<?> clazz) {
330 		for (@SuppressWarnings("unchecked")
331 		Iterator<MModelElement> i = pack.getOwnedElements().iterator(); i.hasNext();) {
332 			MModelElement elem = i.next();
333 
334 			if (name.indexOf("::") > 0 && getQualifiedName(elem).equals(name)
335 					&& clazz.isInstance(elem))
336 				return elem;
337 			else if (name.equals(elem.getName()) && clazz.isInstance(elem))
338 				return elem;
339 			else if (elem instanceof MPackage) {
340 				MModelElement found = searchElement((MPackage) elem, name, clazz);
341 
342 				if (found != null)
343 					return found;
344 			}
345 		}
346 
347 		return null;
348 	}

364 	/**
365 	 * dump the xmi result
366 	 * @throws IOException
367 	 * @throws ru.novosoft.uml.xmi.IncompleteXMIException
368 	 */
369 	@Override
370 	public void dump() throws IOException,
371 			ru.novosoft.uml.xmi.IncompleteXMIException {
372 		XMIWriter writer;
373 		PrintWriter outwriter = null;
374 		if (getDumpPath() == null || "-".equals(getDumpPath())) {
375 			outwriter = new PrintWriter(System.out);
376 			writer = new XMIWriter(model, outwriter);
377 		} else {
378 			writer = new XMIWriter(model, getDumpPath());
379 		}
380 		writer.gen();
381 		writer.close();
382 		if (outwriter!=null) {
383 			outwriter.close();
384 		}
385 	}

401 	/**
402 	 * get a qualifiedName for the given model element
403 	 * @param modelElement 
404 	 * @return - the qualified name for a model element
405 	 */
406 	private static String getQualifiedName(MModelElement modelElement) {
407 		StringBuffer buffer = new StringBuffer();
408 
409 		buffer.append(modelElement.getName());
410 		while (modelElement.getNamespace() != null
411 				&& !(modelElement.getNamespace() instanceof MModel)) {
412 			modelElement = modelElement.getNamespace();
413 			buffer.insert(0, "::");
414 			buffer.insert(0, modelElement.getName());
415 		}
416 		return buffer.toString();
417 	}
