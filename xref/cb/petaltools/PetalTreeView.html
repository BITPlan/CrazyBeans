14  package cb.petaltools;
15  
16  import java.awt.event.WindowAdapter;
17  import java.awt.event.WindowEvent;
18  
19  import javax.swing.JFrame;
20  import javax.swing.JPanel;
21  import javax.swing.JScrollPane;
22  import javax.swing.JTree;
23  
24  import cb.parser.PetalParser;
25  import cb.petal.PetalFile;
26  import cb.util.PetalTree;
27  
28  /**
29   * show a tree view of the given petal file
30   * @author wf
31   *
32   */
33  public class PetalTreeView {
34  	
35  	private PetalParser parser;
36  	private JFrame frame;
37  	
38  	/**
39  	 * @return the parser
40  	 */
41  	public PetalParser getParser() {
42  		return parser;
43  	}
44  
45  	/**
46  	 * @return the frame
47  	 */
48  	public JFrame getFrame() {
49  		return frame;
50  	}
51  
52  	/**
53  	 * show the tree view for the given model FileName
54  	 * @param modelFileName
55  	 * @throws InterruptedException 
56  	 */
57  	public void showTree(String modelFileName) throws InterruptedException {
58  		parser=PetalParser.createParser(modelFileName);
59  		
60  		PetalFile tree = parser.parse();
61  		showTree(tree,null);
62  	}
63  	
64  	/**
65  	 * show the given Tree
66  	 * @param tree - the tree to view
67  	 * @param lock - the Object to lock with (if any)
68  	 * @throws InterruptedException 
69  	 */
70  	public void showTree(PetalFile tree, Object lock) throws InterruptedException {
71  		JTree jtree = new PetalTree(tree);
72  		frame = new JFrame();
73  		frame.setTitle(tree.getModelName());
74  		JPanel panel = new JPanel();
75  		panel.setLayout(new java.awt.BorderLayout());
76  		frame.setContentPane(panel);
77  
78  		JScrollPane scroll = new JScrollPane();
79  		scroll.setPreferredSize(new java.awt.Dimension(400, 500));
80  		scroll.setViewportView(jtree);
81  
82  		panel.add(scroll, java.awt.BorderLayout.CENTER);
83  		frame.setDefaultCloseOperation(JFrame.HIDE_ON_CLOSE);
84  		frame.pack();
85  		frame.setVisible(true);
86  		// shall we keep the tree visible?
87  		if (lock!=null) {
88  			dowait(lock);
89  		}
90  	}
91  	
92  	/**
93  	 * wait on the given lock
94  	 * @param lock
95  	 * @throws InterruptedException 
96  	 */
97  	public void dowait(final Object lock) throws InterruptedException {
98  		Thread t = new Thread() {
99        public void run() {
100           synchronized(lock) {
101               while (frame.isVisible())
102                   try {
103                       lock.wait();
104                   } catch (InterruptedException e) {
105                       e.printStackTrace();
106                   }
107               System.out.println("Working now");
108           }
109       }
110   };
111   t.start();
112 
113   frame.addWindowListener(new WindowAdapter() {
114 
115       @Override
116       public void windowClosing(WindowEvent arg0) {
117           synchronized (lock) {
118               frame.setVisible(false);
119               lock.notify();
120           }
121       }
122 
123   });
124 
125   t.join();
126 }
127 	}
