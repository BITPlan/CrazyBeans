15  package cb.petaltools;
16  
17  import java.util.HashMap;
18  import java.util.Map;
19  import java.util.regex.Matcher;
20  import java.util.regex.Pattern;
21  
22  /**
23   * Path Map utilities
24   * @author moroff
25   *
26   */
27  public class Utils {
28  
29    /**
30     * create a PathMap for the given args
31     * @param args
32     * @return the pathmap created
33     */
34  	public static Map<String,String> createPathMap(String[] args) {
35  		return createPathMap(args, 1);
36  	}
37  	
38  	/**
39  	 * create a path map from the given arguments
40  	 * @param args e.g. $A=APath $B=BPath
41  	 * @param startingArg - where to start e.g. 0
42  	 * @return - a map $A->APath, $B->BPath
43  	 */
44  	public static Map<String,String> createPathMap(String[] args, int startingArg) {
45  		Map<String,String> pathMap = new HashMap<String,String>();
46  	    Pattern pattern = Pattern.compile("(\\$\\w+)=(.+)");
47  	    for ( int i=startingArg; i<args.length; ++i) {
48  	    	Matcher matcher = pattern.matcher(args[i]);
49  	    	if ( matcher.find() ) {
50  		    	String key = matcher.group(1);
51  		    	String path = matcher.group(2);
52  		    	pathMap.put(key,path);
53  	    	}
54  	    }
55  		return pathMap;
56  	}
57  }
