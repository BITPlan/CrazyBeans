3   /**
4    * Denote that a petal object may be extended, i.e. classes and use cases.
5    *
6    * @version $Id: Inheritable.java,v 1.2 2001/07/30 15:50:33 dahm Exp $
7    * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
8    */
9   public abstract class Inheritable extends AccessObject {
10    protected Inheritable(PetalNode parent, String name, java.util.Collection params) {
11      super(parent, name, params);
12    }
13  
14    protected Inheritable(String name) {
15      super(name);
16    }
17  
18    /** @return list of InheritanceRelationship objects
19     */
20    public List getSuperclassList() {
21      return (List)getProperty("superclasses");
22    }
23  
24    /** Set list of InheritanceRelationship objects
25     */
26    public void setSuperclassList(List c) {
27      defineProperty("superclasses", c);
28    }
29  
30    /** Add super classifier of this use case/class, i.e. add InheritanceRelationship to
31     * "superclasses" list.
32     * @return implicitly created relationship object
33     */
34    public InheritanceRelationship addSuperClassifier(Inheritable clazz) {
35      InheritanceRelationship rel =
36        cb.util.PetalObjectFactory.getInstance().createInheritanceRelationship(this, clazz);
37  
38      rel.init(); // Parent is already set
39  
40      addToList("superclasses", "inheritance_relationship_list", rel);
41      return rel;
42    }
43  }
