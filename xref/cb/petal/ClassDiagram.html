1   package cb.petal;
2   import java.util.Arrays;
3   import java.util.Collection;
4   import java.util.HashSet;
5   
6   /**
7    * Represents ClassDiagram object contained within a LogicalCategory
8    * object. Items to the view can be added with the addToView()
9    * methods which call init() on the given Petal Object after adding
10   * them to the items list.
11   *
12   * @version $Id: ClassDiagram.java,v 1.17 2001/07/19 12:40:40 dahm Exp $
13   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
14   */
15  public class ClassDiagram extends Diagram {
16    static final long serialVersionUID = 8346233868380071759L;
17  
18    /**
19     * construct a class diagram from the given collection of params
20     * @param parent
21     * @param params
22     */
23    public ClassDiagram(PetalNode parent, Collection params) {
24      super(parent, "ClassDiagram", params);
25    }
26  
27    public ClassDiagram() {
28      super("ClassDiagram");
29    }
30  
31    /** Adds a note view to the presentation view, sets location and tags
32     * and calls init().
33     */
34    public void addToView(NoteView view) {
35      int index = addToViewsList(view);
36  
37      view.setLocation(new Location(getX(), getY()));
38  
39      // Don't set location on labels ...
40      ItemLabel label = view.getLabel();
41  
42      if(label != null) {
43        label.setParentView(new Tag(index));
44        label.setLocation(view.getLocation());
45      }
46    }
47  
48    /** Attach note to some already added view object.
49    */
50    public void addAttachView(NoteView from, View to) {
51      AttachView view = cb.util.PetalObjectFactory.getInstance().createAttachView();
52      int index = addToViewsList(view);
53      view.setSupplier(new Tag(from.getTag()));
54      view.setClient(new Tag(to.getTag()));
55    }
56  
57    private static HashSet set = new HashSet(Arrays.asList(new java.lang.Object[] {
58      ClassView.class}));
59  
60    protected View searchView(String qual_name) {
61      return searchView(qual_name, set);
62    }
63  
64    /** Class and super class and the according views must have been added
65     * to the model already. Sets tag as well as client and supplier tags.
66     */
67    public void addToView(InheritView view) {
68      view.setParent(this);
69      addRelationship(view, (InheritanceRelationship)view.getReferencedObject());
70    }
71  
72    /** Classes and the according views must have been added
73     * to the model already. Sets tag as well as client and supplier tags.
74     */
75    public void addToView(UsesView view) {
76      view.setParent(this);
77      addRelationship(view, (UsesRelationship)view.getReferencedObject());
78    }
79  
80    /** Class and interface and the according views must have been added
81     * to the model already. Sets tag as well as client and supplier tags.
82     */
83    public void addToView(RealizeView view) {
84      view.setParent(this);
85      addRelationship(view, (RealizeRelationship)view.getReferencedObject());
86    }
87  
88    public void accept(Visitor v) {
89      v.visit(this);
90    }
91  }
