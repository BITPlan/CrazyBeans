133   /** Calculate size for classview by applying some heuristics
134    */
135   protected Dimension getClassViewSize(ClassView view) {
136     cb.petal.Class clazz   = (cb.petal.Class)view.getReferencedObject();
137     int   max_columns = clazz.getNameParameter().length();
138     int   rows    = 3;
139 
140     String stereo = clazz.getStereotype();
141 
142     if(stereo != null) {
143       rows++;
144 
145       int columns = 4 + stereo.length();
146       if(columns > max_columns)
147         max_columns = columns;
148     }
149 
150     for(Iterator i=clazz.getOperations().iterator(); i.hasNext(); ) {
151       Operation op = (Operation)i.next();
152 
153       int columns = op.getNameParameter().length() + 2;
154 
155       stereo = op.getStereotype();
156       if(stereo != null)
157         columns += stereo.length() + 4;
158 
159       if(columns > max_columns)
160         max_columns = columns;
161 
162       rows++;
163     }
164 
165     for(Iterator i=clazz.getClassAttributes().iterator(); i.hasNext(); ) {
166       ClassAttribute attr = (ClassAttribute)i.next();
167 
168       int columns = attr.getNameParameter().length() + 2 + attr.getType().length();
169 
170       stereo = attr.getStereotype();
171       if(stereo != null)
172         columns += stereo.length() + 4;
173 
174       if(columns > max_columns)
175         max_columns = columns;
176 
177       rows++;
178     }
179 
180     return new Dimension(max_columns * 30, rows * 50);
181   }

183   /** Adds a class view to the presentation view, sets location and tags
184    * and calls init().
185    */
186   public void addToView(ClassView view) {
187     int index = addToViewsList(view);
188 
189     view.setLocation(new Location(getX(), getY()));
190 
191     cb.petal.Class clazz = (cb.petal.Class)view.getReferencedObject();
192 
193     // This is just heuristic!
194     if(!clazz.isActor()) {
195       Dimension dim = getClassViewSize(view);
196       view.setWidth(dim.width);
197       view.setHeight(dim.height);
198     }
199 
200     // Don't set location on labels ...
201     ItemLabel label = view.getLabel();
202 
203     if(label != null)
204       label.setParentView(new Tag(index));
205 
206     label = view.getStereotype();
207 
208     if(label != null)
209       label.setParentView(new Tag(index));
210   }

212   protected void addRelationship(RelationshipView view, Relationship rel) {
213     int    index       = addToViewsList(view);
214     String clazz       = ((PetalObject)rel.getParent()).getQualifiedName();
215     String super_class = ((PetalObject)rel.getReferencedObject()).getQualifiedName();
216     int client         = searchView(clazz).getTag();
217     int supplier       = searchView(super_class).getTag();
218 
219     view.setClient(new Tag(client));
220     view.setSupplier(new Tag(supplier));
221 
222     ItemLabel label = view.getLabel();
223     if(label != null) {
224       label.setParent(view);
225       label.setParentView(new Tag(((Tagged)view).getTag()));
226     }
227 
228     SegLabel stereo = view.getStereotype();
229     if(stereo != null) {
230       stereo.setParent(view);
231       stereo.setTag(getRoot().getNewTag());
232       stereo.setParentView(new Tag(((Tagged)view).getTag()));
233     }
234   }

236   protected abstract View searchView(String qual_name);
237 
238   protected View searchView(String qual_name, HashSet classes) {
239     for(java.util.Iterator i = getItems().getElements().iterator(); i.hasNext();) {
240       View view = (View)i.next();
241 
242       if(classes.contains(view.getClass())) {
243 	Qualified view2 = (Qualified)view;
244 
245 	if(view2.getQualifiedNameParameter().equals(qual_name))
246 	  return view;
247       }
248     }
249 
250     throw new RuntimeException("No view found for " + qual_name);
251   }

253   /** Add a association view to the presentation view, set location and tags
254    * and call init(). Automatically adds AssocAttachView, if the association has
255    * an association class.
256    */
257   public void addToView(AssociationViewNew view) {
258     int index = addToViewsList(view);
259 
260     //view.defineProperty("location", new Location(getX(), getY()));
261     setupLabels(view);
262 
263     RoleView first  = view.getFirstRoleView();
264     RoleView second = view.getSecondRoleView();
265 
266     first.setTag(getRoot().getNewTag());
267     second.setTag(getRoot().getNewTag());
268 
269     Tag tag = new Tag(index);
270     first.setParentView(tag);
271     first.setClient(tag);
272     second.setParentView(tag);
273     second.setClient(tag);
274 
275     Association assoc = (Association)view.getReferencedObject();
276     Role role1 = assoc.getFirstRole();
277     Role role2 = assoc.getSecondRole();
278 
279     View clazz1 = searchView(role1.getSupplier());
280     View clazz2 = searchView(role2.getSupplier());
281 
282     first.setSupplier(new Tag(clazz1.getTag()));
283     second.setSupplier(new Tag(clazz2.getTag()));
284 
285     setupLabels(first);
286     setupLabels(second);
287 
288     if(assoc.getAssociationClass() != null) {
289       AssocAttachView view2 = cb.util.PetalObjectFactory.getInstance().createAssocAttachView();
290       addToViewsList(view2);
291       view2.setClient(tag);
292       View clazz = searchView(assoc.getAssociationClass().getQualifiedName());
293       view2.setSupplier(new Tag(clazz.getTag()));
294     }
295   }

297   private void setupLabels(SegLabeled view) {
298     ((View)view).setLocation(new Location(getX(), getY()));
299 
300     for(Iterator i = ((PetalObject)view).getProperties("label").iterator();
301     	i.hasNext(); ) {
302       SegLabel label = (SegLabel)i.next();
303 
304       label.setParent((PetalObject)view);
305       label.setTag(getRoot().getNewTag());
306       label.setParentView(new Tag(((Tagged)view).getTag()));
307     }
