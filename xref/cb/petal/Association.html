1   package cb.petal;
2   import java.util.*;
3   
4   /**
5    * Represents Association object
6    *
7    * @version $Id: Association.java,v 1.15 2001/08/01 14:26:56 dahm Exp $
8    * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
9    */
10  public class Association extends AccessObject {
11    static final long serialVersionUID = 2685090660257573719L;
12  
13    public Association(PetalNode parent, Collection params) {
14      super(parent, "Association", params);
15    }
16  
17    public Association() {
18      super("Association");
19    }
20  
21    /** @return Class or UseCase
22     */
23    public QuidObject getFirstClient() {
24      return getFirstRole().getReferencedObject();
25    }
26  
27    /** @return Class or UseCase
28     */
29    public QuidObject getSecondClient() {
30      return getSecondRole().getReferencedObject();
31    }
32  
33    /** An association contains exactly two roles. Get the first one.
34     */
35    public Role getFirstRole() {
36      return (Role)getRoles().getElements().get(0);
37    }
38  
39    /** An association contains exactly two roles. Get the second one.
40     */
41    public Role getSecondRole() {
42      return (Role)getRoles().getElements().get(1);
43    }
44  
45    /** Register this association internally, i.e. associate it with the
46     * attached classes. So the classes can look up the associations
47     * related to them, too.<p>
48     * If this association has an association class, set its
49     * isAssociationClass() flag.
50     */
51    public void init() {
52      super.init();
53      getRoot().registerAssociation(this);
54  
55      Class clazz = getAssociationClass();
56      if(clazz != null)
57        clazz.isAssociationClass(true);
58    }
59  
60    public Class getAssociationClass() {
61      String s = getPropertyAsString("AssociationClass");
62  
63      if(s != null) {
64        Class clazz = getRoot().getClassByQualifiedName(s);
65  
66        if(clazz == null)
67  	System.err.println("Warning: Could not find association class " + s +
68  			   " (forward declaration?)");
69  
70        return clazz;
71      } else
72        return null;
73    }
74  
75    public void setAssociationClass(Class o) {
76      setAssociationClass(o.getQualifiedName());
77    }
78  
79    /** Set association class via its fully qualified name like
80     * "Logical View::University::Period".
81     */
82    public void setAssociationClass(String qual_name) {
83      defineProperty("AssociationClass", qual_name);
84    }
85  
86    public List getRoles() {
87      return (List)getProperty("roles");
88    }
89  
90    public void setRoles(List o) {
91      defineProperty("roles", o);
92    }
93  
94    public void accept(Visitor v) {
95      v.visit(this);
96    }
97  }
