12  package cb.petal;
13  
14  import java.util.*;
15  
16  /**
17   * Super class for all petal objects that have access qualifiers, i.e.,
18   * "Public", "Protected", "Private", or "Implementation" (whatever that means, probably
19   * it reads "language dependent")
20   *
21   * @version $Id: AccessObject.java,v 1.4 2001/07/09 07:48:52 dahm Exp $
22   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
23   */
24  public abstract class AccessObject extends QuidObject
25    implements AccessQualified, Named, StereoTyped, Documented
26  {
27    static final long serialVersionUID = -1442259881847075234L;
28  
29    protected AccessObject(PetalNode parent, String name, Collection params) {
30      super(parent, name, params);
31    }
32  
33    protected AccessObject(String name) {
34      super(name);
35    }
36  
37    public void setNameParameter(String o) {
38      params.set(0, o);
39    }
40  
41    public String getNameParameter() {
42      return (String)params.get(0);
43    }
44  
45    public String getDocumentation() {
46      return getPropertyAsString("documentation");
47    }
48  
49    public void setDocumentation(String o) {
50      defineProperty("documentation", o);
51    }
52  
53    public String getStereotype() {
54      return getPropertyAsString("stereotype");
55    }
56  
57    public void setStereotype(String c) {
58      defineProperty("stereotype", c);
59    }
60  
61    public String getExportControl() {
62      return getPropertyAsString("exportControl");
63    }
64  
65    public void setExportControl(String o) {
66      defineProperty("exportControl", o);
67    }
68  
69    public boolean isPublic() {
70      return "Public".equals(getExportControl());
71    }
72  
73    public boolean isProtected() {
74        return "Protected".equals(getExportControl());
75    }
76  
77    public boolean isPrivate() {
78        return "Private".equals(getExportControl());
79    }
80  }
