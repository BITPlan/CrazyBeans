6   /**
7    * The Logical category contains class diagrams. The interesting part
8    * are the two lists "logical_models" and "logical_presentations"
9    * where the former contains the data model and the latter contains
10   * class diagrams. One can add items to the model with addToModel()
11   * methods. Items to the view can be added with the addToView()
12   * methods in ClassDiagram. Both types of methods call init() on the
13   * given Petal Object after adding them to the appropriate list.
14   *
15   * @version $Id: LogicalCategory.java,v 1.9 2001/07/17 14:38:36 dahm Exp $
16   * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
17   * @see ClassDiagram
18   */
19  public class LogicalCategory extends ClassCategory {
20    static final long serialVersionUID=421992761725074833L;
21  
22    /**
23     * create the LogicalCategory - which is the root of the ClassCategory tree
24     * @param parent
25     */
26    public LogicalCategory(PetalObject parent) {
27      super(parent, "Logical View");
28    }
29  
30    public LogicalCategory() {
31      super(null, "Logical View");
32    }
33  
34    public String getSubsystem() {
35      return getPropertyAsString("subsystem");
36    }
37  
38    public void setSubsystem(String o) {
39      defineProperty("subsystem", o);
40    }
41  
42    public void addToModel(LogicalCategory cat) {
43      add(cat);
44    }
45  
46    public void removeFromModel(LogicalCategory cat) {
47      remove(cat);
48    }
49  
50    /** Typically every Logical view has just one "Main" class diagram
51     */
52    public ClassDiagram getFirstClassDiagram() {
53      return (ClassDiagram)lookupDiagram(ClassDiagram.class);
54    }
55  
56    public void accept(Visitor v) {
57      v.visit(this);
58    }
59  }
