1   package cb.petal;
2   import java.text.NumberFormat;
3   import java.util.*;
4   
5   /**
6    * Floating point literal (in fact a double).
7    *
8    * @version $Id: FloatLiteral.java,v 1.10 2001/07/09 07:48:52 dahm Exp $
9    * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
10   */
11  public class FloatLiteral extends Literal {
12    static final long serialVersionUID=1505488095051523558L;
13  
14    private double value;
15    private static NumberFormat format;
16  
17    static {
18      format = NumberFormat.getInstance(Locale.ENGLISH);
19      format.setMaximumFractionDigits(6);
20      format.setMinimumFractionDigits(6);
21    }
22  
23    public FloatLiteral(double value) {
24      super("<float>");
25      this.value = value;
26    }
27  
28    public double getValue()         { return value; }
29    public void   setValue(double d) { value = d; }
30  
31    public java.lang.Object getLiteralValue() {
32      return new Float(value);
33    }
34  
35    public String toString() {
36      return format.format(value);
37    }
38    
39    public void accept(Visitor v) {
40      v.visit(this);
41    }
42  
43    public boolean equals(java.lang.Object o) {
44      return (o instanceof FloatLiteral) && (((FloatLiteral)o).value == this.value);
45    }
46  }
