86  	/**
87  	 * Find diagram by given class
88  	 */
89  	protected java.lang.Object lookupDiagram(java.lang.Class clazz) {
90  		List list = getLogicalPresentations();
91  
92  		if ((list == null) || (list.size() == 0))
93  			return null;
94  		else {
95  			for (Iterator i = list.getElements().iterator(); i.hasNext();) {
96  				java.lang.Object o = i.next();
97  
98  				if (o.getClass() == clazz)
99  					return o;
100 			}
101 		}
102 
103 		return null;
104 	}
105
