3   /**
4    * Super class for literals like int, String, value, etc.
5    *
6    * @version $Id: Literal.java,v 1.7 2001/07/09 07:48:52 dahm Exp $
7    * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
8    */
9   public abstract class Literal implements PetalNode {
10    static final long serialVersionUID = 3238150669982522033L;
11    private String kind;
12  
13    protected Literal(String kind) {
14      this.kind = kind;
15    }
16  
17    public String getKind() { return kind; }
18    public int getChildCount() { return 0; }
19  
20    /** @return object corresponding to the usual Java hierarchy, e.g., an Integer for
21     * an IntegerLiteral.
22     */
23    public abstract java.lang.Object getLiteralValue();
24  
25    public java.lang.Object clone() {
26      try {
27        return super.clone();
28      } catch(CloneNotSupportedException e) {
29        return null;
30      }
31    }
32  }
