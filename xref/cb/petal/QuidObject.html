1   package cb.petal;
2   
3   import java.util.*;
4   
5   /**
6    * Super class for all petal objects that have a "quid" property
7    * defined, i.e. a globally unique identifier.
8    *
9    * @version $Id: QuidObject.java,v 1.6 2001/07/09 07:48:52 dahm Exp $
10   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
11   */
12  public abstract class QuidObject extends PetalObject {
13    static final long serialVersionUID = -1442259881047075234L;
14  
15    /**
16     * create a QuidObject for the given parent with the given name and the
17     * given collection of params
18     * @param parent
19     * @param name
20     * @param params
21     */
22    protected QuidObject(PetalNode parent, String name, Collection params) {
23      super(parent, name, params);
24    }
25  
26    protected QuidObject(String name) {
27      super(name);
28    }
29  
30    static String getQuid(QuidObject obj) {
31      return PetalObject.getPropertyAsString(obj, "quid");
32    }
33  
34    /** 
35     * Registers this object.
36     */
37    public void init() {
38      getRoot().registerQuidObject(this);
39    }
40  
41    public void setQuid(String quid) {
42      defineProperty("quid", quid);
43    }
44  
45    public String getQuid() {
46      return getQuid(this);
47    }
48  
49    public long getQuidAsLong() {
50      return Long.parseLong(getQuid(), 16);
51    }
52  
53    public void setQuidAsLong(long quid) {
54      defineProperty("quid", Long.toHexString(quid).toUpperCase());
55    }
56  }
