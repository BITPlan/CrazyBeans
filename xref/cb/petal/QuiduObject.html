5   /**
6    * Super class for all quid objects that also have a quidu property
7    * defined.
8    *
9    * @version $Id: QuiduObject.java,v 1.5 2001/06/22 09:10:36 dahm Exp $
10   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
11   */
12  public abstract class QuiduObject extends QuidObject implements HasQuidu {
13    protected QuiduObject(PetalNode parent, String name, Collection params) {
14      super(parent, name, params);
15    }
16  
17    protected QuiduObject(String name) {
18      super(name);
19    }
20  
21    static String getQuidu(HasQuidu obj) {
22      return PetalObject.getPropertyAsString((PetalObject)obj, "quidu");
23    }
24  
25    public void setQuidu(String quid) {
26      defineProperty("quidu", quid);
27    }
28  
29    public String getQuidu() {
30      return getQuidu(this);
31    }
32  
33    public long getQuiduAsLong() {
34      return Long.parseLong(getQuidu(), 16);
35    }
36  
37    public void setQuiduAsLong(long quid) {
38      defineProperty("quidu", Long.toHexString(quid).toUpperCase());
39    }
40  
41    public final QuidObject getReferencedObject() {
42      return getRoot().getReferencedObject(this);
43    }
44  }
