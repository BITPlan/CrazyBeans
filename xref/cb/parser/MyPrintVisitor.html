12  package cb.parser;

14  import java.util.Iterator;
15  
16  import cb.petal.Association;
17  import cb.petal.HasQuidu;
18  import cb.petal.Named;
19  import cb.petal.PetalFile;
20  import cb.petal.PetalNode;
21  import cb.petal.PetalObject;
22  import cb.petal.QuidObject;
23  import cb.petal.Role;
24  
25  /**
26   * (Experimental) Just prints some information about the traversed class.
27   *
28   * @version $Id: MyPrintVisitor.java,v 1.7 2001/11/01 15:56:49 dahm Exp $
29   * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
30   */
31  public class MyPrintVisitor extends PrintVisitor {
32    public MyPrintVisitor(PrintStream out) {
33      super(out);
34    }
35  
36    public void visitObject(PetalObject obj) {
37      PetalNode node = obj.getProperty("quidu");
38  
39      if(node != null) {
40        if(!(obj instanceof HasQuidu))
41  	System.err.println(obj.getName() + " has quidu " + node);
42      } else if(obj instanceof HasQuidu) {
43        System.err.println(obj.getName() + " has NO quidu ");
44      }
45  
46  
47      super.visitObject(obj);
48    }
49  
50    public void visit(cb.petal.Class obj) {
51      cb.petal.Class super_ = obj.getSuperclass();
52      java.util.List list   = obj.getAssociations();
53  
54      System.err.println(obj.getQualifiedName());
55  
56      if(super_ != null) {
57        System.err.println("Super class of " + obj.getNameParameter() + ":" +
58  			 super_.getNameParameter());
59      } else if(list != null) {
60  
61        System.err.print("Associations of " + obj.getNameParameter() + ":");
62  
63        for(Iterator i = list.iterator(); i.hasNext(); ) {
64  	Association a = (Association)i.next();
65  	System.err.println(((Named)a.getFirstClient()).getNameParameter() + " <-> " +
66  			   ((Named)a.getSecondClient()).getNameParameter());
67  
68  	cb.petal.Class clazz = a.getAssociationClass();
69  	if(clazz != null)
70  	  System.err.println("ASSOCIATIONCLASS:" + clazz.getNameParameter());
71        }
72      }
73  
74      visitObject(obj);
75    }
76  
77    public void visit(Association obj) {
78      java.util.List list = obj.getRoles().getElements();
79  
80      Role role1   = (Role)list.get(0);
81      Role role2   = (Role)list.get(1);
82  
83      QuidObject class1 = obj.getFirstClient();
84      QuidObject class2 = obj.getSecondClient();
85  
86      String name1 = role1.getNameParameter();
87      String name2 = role2.getNameParameter();
88      String card1 = role1.getCardinality();
89      String card2 = role2.getCardinality();
90  
91      System.err.println("Association named " + obj.getNameParameter() + " between:\n" +
92  		       ((Named)class1).getNameParameter() +
93  		       "(" + name1 + ":" + card1 +
94  		       ":" + role1.isNavigable() + ":" + role1.isAggregate() +
95  		       ")" +
96  		       "\n" +
97  		       ((Named)class2).getNameParameter() +
98  		       "(" + name2 + ":" + card2 +
99  		       ":" + role2.isNavigable() + ":" + role2.isAggregate() +
100 		       ")"
101 		       );
102     visitObject(obj);
103   }
104 
105   public static void main(String[] args) {
106     PetalFile tree = PetalParser.parse(args);
107     tree.accept(new MyPrintVisitor(System.out));
108   }
109 }
