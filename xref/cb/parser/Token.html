5   public class Token {
6     public static final Token LPAREN = new Token(Lexer.LPAREN, "(", -1);

8     public static final Token COMMA  = new Token(Lexer.COMMA, ",", -1);
9     public static final Token TRUE   = new Token(Lexer.BOOLEAN, "TRUE", -1);
10    public static final Token FALSE  = new Token(Lexer.BOOLEAN, "FALSE", -1);
11    public static final Token EOF    = new Token(Lexer.EOF, "<EOF>", -1);
12  
13    private Token() {}
14  
15    private Token(int kind, String image, int line) {
16      init(kind, image, line);
17    }
18  
19    private void init(int kind, String image, int line) {
20      this.kind  = kind;
21      this.image = image;
22      this.line  = line;
23    }
24    
25    private static Stack stack = new Stack();
26  
27    public static Token createToken(int kind, String image, int line) {
28      Token t;
29  
30      if(stack.isEmpty())
31        t = new Token();
32      else
33        t = (Token)stack.pop();
34  
35      t.init(kind, image, line);
36      return t;
37    }
38  
39    public static void dispose(Token t) {
40      if(t != null)
41        stack.push(t);
42    }
43  
44    public int    kind;
45    public String image;
46    public int    line;
47  }
