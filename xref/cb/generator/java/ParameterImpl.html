8    * Simple representation of a method parameter.

10   * @version $Id: ParameterImpl.java,v 1.2 2001/06/27 10:26:03 dahm Exp $

13  public class ParameterImpl extends NodeImpl implements cb.generator.java.Parameter {
14    private   String  type;
15    protected cb.petal.Parameter parameter;
16  
17    /**
18     * construct me
19     */
20    public ParameterImpl() {
21    		super(null); // TODO can not get id from parameters since cb.petal.Parameter is not a quid object
22    }
23  
24    public void               setParameter(cb.petal.Parameter p) { super.setDocumentedObject(p);;parameter = p; }
25    public cb.petal.Parameter getParameter()                     { return parameter; }
26  
27    public void   setType(String p) { type = p; }
28    public String getType() { return type; }
29  
30    /** Default implementation prints Java code
31     */
32    public void dump(PrintWriter stream) {
33      print(stream, "", getAccess(), " ");
34      stream.print(getType() + " " + getName());
35    }
36  
37    public boolean equals(Object o) {
38      if(o instanceof Parameter) {
39        Parameter p = (Parameter)o;
40  
41        return getType().equals(p.getType()) && getName().equals(p.getName());
42      } else
43        return false;
44    }
45  }
