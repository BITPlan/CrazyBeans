12  package cb.generator.java;
13  
14  import java.util.Collection;
15  
16  /**
17   * Represents a class consisting of fields and methods, should be useful
18   * for Java and C++.
19   *
20   * @version $Id: Class.java,v 1.4 2001/06/27 10:26:03 dahm Exp $
21   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
22   */
23  public interface Class extends Node {
24    public void   addPrefixCode(String c); // Inserted before class itself, e.g. imports
25    public void   addSuperClass(String c);
26    public void   removeSuperClass(String f);
27    public void   addImplementedInterface(String c);
28    public void   removeImplementedInterface(String f);
29    public void   addField(Field f);
30    public void   removeField(Field f);
31    public void   addMethod(Method m);
32    public void   removeMethod(Method m);
33    public void setPackage(Package p);
34    public Package getPackage();
35    public boolean isInterface();
36    public Collection<Field> getFields();
37    public Collection<Method> getMethods();
38    public String getQualifiedName();
39  }
