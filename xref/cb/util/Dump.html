37   * @version $Id: Dump.java,v 1.10 2001/11/01 15:56:49 dahm Exp $

40  public class Dump {
41    private static class Traverser extends DescendingVisitor {
42      public void visit(cb.petal.Class obj) {
43        if (obj.getNameParameter().equals("SomeClass")) {
44          obj.setParent(null);
45  
46          dump("Class", obj);
47        }
48  
49        visitObject(obj);
50      }
51  
52      public void visit(ClassView obj) {
53        if (obj.getQualifiedNameParameter().equals(
54            "Logical View::templates::SomeClass")) {
55          obj.setParent(null);
56          dump("ClassView", obj);
57        } else if (obj.getQualifiedNameParameter().equals(
58            "Logical View::templates::Actor")) {
59          obj.setParent(null);
60          dump("ActorView", obj);
61        } else if (obj.getQualifiedNameParameter().equals(
62            "Logical View::templates::Interface")) {
63          obj.setParent(null);
64          dump("StereotypeView", obj);
65        }
66  
67        visitObject(obj);
68      }
69  
70      public void visit(UseCaseView obj) {
71        if (obj.getQualifiedNameParameter().equals("Use Case View::Examination")) {
72          obj.setParent(null);
73          dump("UseCaseView", obj);
74        } else if (obj.getQualifiedNameParameter().equals(
75            "Use Case View::Lecture")) {
76          obj.setParent(null);
77          dump("UseCaseStereotypeView", obj);
78        }
79      }
80  
81      public void visit(InheritView obj) {
82        if (obj.getQuidu().equals("3B29C6640186")) {
83          obj.setParent(null);
84          dump("InheritView", obj);
85        }
86      }
87  
88      public void visit(RealizeView obj) {
89        if (obj.getQuidu().equals("3B29C6F203D4")) {
90          obj.setParent(null);
91          dump("RealizeView", obj);
92        }
93  
94        visitObject(obj);
95      }
96  
97      public void visit(UsesView obj) {
98        if (obj.getQuidu().equals("3B29C88B0014")) {
99          obj.setParent(null);
100         dump("UsesView", obj);
101       }
102 
103       visitObject(obj);
104     }
105 
106     public void visit(Association obj) {
107       if (obj.getQuid().equals("3B29CAD600E6")) {
108         obj.setParent(null);

110       }
111

114

160     public void visit(SegLabel obj) {
161       if (obj.getLabel().equals("1..*")) {
162         obj.setParent(null);
163         obj.removeProperty("anchor");
164         obj.removeProperty("anchor_loc");
165         dump("SegLabel", obj);
166       }
167 
168       visitObject(obj);
169     }

173         obj.setParent(null);
174         dump("Operation", obj);
175       }
176 
177       visitObject(obj);
178     }

181   /**
182    * dump the given node to a file with the given name in the template directory
183    * 
184    * @param name
185    * @param node
186    */
187   static void dump(String name, PetalNode node) {
188     try {
189       File file = new File("src/main/resources/templates" + File.separatorChar + name + ".ser");
190       ObjectOutputStream s = new ObjectOutputStream(new FileOutputStream(file));
191       s.writeObject(node);
192       s.close();
193     } catch (Exception e) {
194       e.printStackTrace();
195       return;
196     }
197   }

199   /**
200    * create the template files
201    * @param args
202    */
203   public static void main(String[] args) {
204     PetalFile tree = null;
205 
206     try {
207       tree = PetalParser.createParser(
208           "examples" + File.separatorChar + "empty.mdl").parse();
209 
210       dump("PetalFile", tree);
211       dump("LogicalCategory", tree.getLogicalCategory());
212       dump("UseCaseCategory", tree.getUseCaseCategory());
213 
214       tree = PetalParser.createParser(
215           "examples" + File.separatorChar + "uni.mdl").parse();
216 
217       Traverser t = new Traverser();
218       tree.accept(t);
219     } catch (Exception e) {
220       e.printStackTrace();
221     }
222   }
