31  /**
32   * Display petal file in JTree. The tooltip text displays the property name, if
33   * the currently selected node is a child of a PetalObject.
34   *
35   * @version $Id: PetalTree.java,v 1.1 2001/07/06 11:09:39 dahm Exp $
36   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
37   */
38  public class PetalTree extends JTree {
39    /**
40     * 
41     */
42    private static final long serialVersionUID = 3162615092990086860L;
43    protected static final ImageIcon PLUS_ICON = getIcon("PLUS.GIF");
44    protected static final ImageIcon MINUS_ICON = getIcon("MINUS.GIF");
45    protected static final ImageIcon OPENED_ICON = getIcon("OPENED.GIF");
46    protected static final ImageIcon CLOSED_ICON = getIcon("CLOSED.GIF");
47    protected static final ImageIcon LEAF_ICON = getIcon("LEAF.GIF");
48  
49    public PetalTree(PetalFile tree) {
50      super(tree);
51      getSelectionModel().setSelectionMode(
52          TreeSelectionModel.SINGLE_TREE_SELECTION);
53      ToolTipManager.sharedInstance().registerComponent(this);
54      putClientProperty("JTree.lineStyle", "Angled");
55      setShowsRootHandles(false);
56      setUI(new MyTreeUI());
57      setCellRenderer(new MyTreeCellRenderer());
58    }
59  
60    private static ImageIcon getIcon(String name) {
61      try {
62        return new ImageIcon(PetalTree.class.getResource("/images/" + name));
63      } catch (Exception e) {
64        System.err.println("Couldn't find image: " + name);
65      }
66  
67      return new ImageIcon();
68    }
69  
70    private static class MyTreeUI extends javax.swing.plaf.metal.MetalTreeUI {
71      public javax.swing.Icon getExpandedIcon() {
72        return MINUS_ICON;
73      }
74  
75      public javax.swing.Icon getCollapsedIcon() {
76        return PLUS_ICON;
77      }
78    }
79  
80    private static class MyTreeCellRenderer extends DefaultTreeCellRenderer {
81      /**
82       * 
83       */
84      private static final long serialVersionUID = 8446162207536434648L;
85  
86      public java.awt.Component getTreeCellRendererComponent(JTree tree,
87          java.lang.Object value, boolean sel, boolean expanded, boolean leaf,
88          int row, boolean hasFocus) {
89  
90        if (value instanceof PetalFile)
91          value = "Root";
92        else if (value instanceof PetalObject) {
93          PetalObject obj = (PetalObject) value;
94          StringBuffer buf = new StringBuffer(obj.getName());
95  
96          for (@SuppressWarnings("rawtypes")
97          Iterator i = obj.getParameterList().iterator(); i.hasNext();)
98            buf.append(" \"" + i.next() + "\"");
99  
100         if (obj instanceof Tagged)
101           buf.append(" " + ((Tagged) obj).getTag());
102 
103         value = buf;
104       } else if (value instanceof List) {
105         value = "list " + ((List) value).getName();
106       } else if ((value instanceof StringLiteral)) {
107         value = '"' + (String) ((StringLiteral) value).getLiteralValue() + '"';
108       } else if ((value instanceof Value)) {
109         value = ((Value) value).getLiteralValue();
110       }
111 
112       super.getTreeCellRendererComponent(tree, value, sel, expanded, leaf, row,
113           hasFocus);
114 
115       if (leaf) {
116         setIcon(LEAF_ICON);
117       } else {
118         if (expanded)
119           setIcon(OPENED_ICON);
120         else
121           setIcon(CLOSED_ICON);
122       }
123 
124       return this;
125     }
126   }
127 
128   public String getToolTipText(java.awt.event.MouseEvent e) {
129     if (e != null) {
130       // int row = getRowForLocation(e.getX(), e.getY());
131       TreePath path = getPathForLocation(e.getX(), e.getY());
132 
133       if ((path != null)) {
134         PetalNode node = (PetalNode) path.getLastPathComponent();
135 
136         path = path.getParentPath();
137 
138         if (path != null) {
139           PetalNode parent = (PetalNode) path.getLastPathComponent();
140 
141           if (parent instanceof PetalObject)
142             return ((PetalObject) parent).getPropertyName(node);
143         }
144       }
145     }
146 
147     return null;
148   }
