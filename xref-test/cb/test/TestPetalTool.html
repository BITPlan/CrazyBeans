32  public class TestPetalTool extends BaseTest {
33  
34    /**
35     * for grabbing stderr
36     */
37    private String errText;
38    private String outText;
39  
40    /**
41     * test the petal tool command line
42     * 
43     * @param args
44     *          - command line arguments
45     * @param expectedExit
46     *          - the expected exit code
47     * @param sleepTime
48     * @throws Exception
49     */
50    public void testPetalTool(String args[], int expectedExit, int sleepTime)
51        throws Exception {
52      PetalTool.testMode = true;
53      PrintStream stdout = System.out;
54      PrintStream stderr = System.err;
55      ByteArrayOutputStream errStream = new ByteArrayOutputStream();
56      ByteArrayOutputStream outStream = new ByteArrayOutputStream();
57      try {
58        System.setOut(new PrintStream(outStream, true));
59        System.setErr(new PrintStream(errStream, true));
60        PetalTool.main(args);
61        Thread.sleep(sleepTime);
62      } catch (InterruptedException e) {
63        // ignore
64      } finally {
65        System.setOut(stdout);
66        System.setErr(stderr);
67      }
68      errText = errStream.toString();
69      outText = outStream.toString();
70      if (debug) {
71        LOGGER.log(Level.INFO, "stderr:\n" + errText);
72        LOGGER.log(Level.INFO, "stdout:\n" + outText);
73      }
74      assertEquals("" + Arrays.toString(args), expectedExit, PetalTool.exitCode);
75    }
76  
77    @Test
78    public void testHelp() throws Exception {
79      String[] args = { "-h" };
80      // debug=true;
81      testPetalTool(args, 1, 0);
82      assertTrue(errText
83          .contains("github: https://github.com/BITPlan/CrazyBeans"));
84      assertTrue(errText.contains("usage:"));
85    }

87    /**
88     * test exporting the example model files to the different formats
89     * 
90     * @throws Exception
91     */
92    @Test
93    public void testExportExamples() throws Exception {
94      for (Example exampleModel : exampleModels) {
95        String exampleModelFilePath = exampleModel.getFilePath();
96        String formats[] = { "xmi", "rose","java" };
97        for (String format : formats) {
98          String[] args = { "--input", exampleModelFilePath, "--format", format, "--source-root", exampleModel.getSrcRoot() };
99          // debug = true;
100         if (debug)
101           LOGGER.log(Level.INFO, format + "-Export for " + exampleModelFilePath
102               + " with " + Arrays.toString(args));
103         debug = false;
104         int expectedExitCode = 0;
105         if ("examples/project.mdl".equals(exampleModelFilePath)
106             && "xmi".equals(format)) {
107           expectedExitCode = 1; // RuntimeException:Can't map access qualifier:
108           // implementation
109         }
110         if ("examples/JDK-12_01.mdl".equals(exampleModelFilePath)) {
111           expectedExitCode = 1; // FIXME - see BaseTest reason for problem
112         }
113         testPetalTool(args, expectedExitCode, 0);
114       }
115     }
