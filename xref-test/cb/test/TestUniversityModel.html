18

25  import org.junit.Test;
26

34  import cb.petal.NoteView;
35  import cb.petal.PetalFile;
36  import cb.petal.RealizeView;

38  import cb.petal.UseCase;

40  import cb.petal.UseCaseDiagram;
41  import cb.petal.UsesRelationship;
42  import cb.petal.UsesView;

44  /**
45   * Create nonsense university model found in examples/uni.mdl from scratch.
46   *
47   * @version $Id: Test1.java,v 1.13 2001/07/19 12:40:40 dahm Exp $
48   * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
49   */
50  public class TestUniversityModel extends BaseTest {
51  	@SuppressWarnings("unchecked")
52  	@Test
53    public void testUniversityModel()  {
54     
55  
56      ///////////////// Create data and view for class diagram /////////////
57  
58      // Create objects via factory
59      PetalFile      model      = factory.createModel();
60      cb.petal.Class person     = factory.createClass("Person");
61      cb.petal.Class prof       = factory.createClass("Professor");
62      cb.petal.Class student    = factory.createClass("Student");
63      cb.petal.Class mentor     = factory.createInterface("Mentor");
64      cb.petal.Class money      = factory.createClass("Money");
65      cb.petal.Class department = factory.createClass("Department");
66      cb.petal.Class uni        = factory.createClass("University");
67      cb.petal.Class period     = factory.createClass("Period");
68  
69      person.setDocumentation("Super class of all beings\nrelevant to the system");
70  
71      /* Create objects on the fly yourself, just for demonstration. Class objects have
72       * a name parameter, i.e. the name is not defined as a property. Therefore we just
73       * set the parameter list to a singleton list just containing the name of the class.
74       */
75      ClassUtility   address = new ClassUtility();
76      address.setParameterList(new ArrayList<String>(Arrays.asList(new String[]{"Address"})));
77  
78      // Its important to give the class a unique id, the getQuid() method ensures that
79      address.setQuid(PetalFile.getQuid());
80  
81      /* Every petal model has a class category "Logical View" and a class category
82       * "Use Case View". Both have a property "logical_models" which refers to a list
83       * of classes, associations, ... They also have a property "logical_presentations",
84       * which contains the views, i.e., diagrams of the model. We start by adding
85       * data to the logical model first.
86       */
87      LogicalCategory cat = model.getLogicalCategory();
88  
89      cat.addToModel(person);
90      cat.addToModel(prof);
91      cat.addToModel(student);
92      cat.addToModel(mentor);
93      cat.addToModel(address);
94      cat.addToModel(money);
95      cat.addToModel(uni);
96      cat.addToModel(department);
97      cat.addToModel(period);
98  
99      /* Create (empty) sub-package just for demonstration, it could contain
100      * further data and view models.
101      */
102     cat.addToModel(factory.createLogicalCategory("Foo - Just a test"));
103 
104     /* Once we've added the classes to the model we can start adding
105      * inheritance relationships and associations between them.
106      */
107     prof.addSuperClass(person);
108     prof.addImplementedInterface(mentor);
109     prof.setStereotype("Actor");
110 
111     /* Add used class and adapt relationship object, we don't need to create the
112      * relationship object ourselves via the factory, but can use the convenience
113      * method in "Class".
114      */
115     UsesRelationship rel = prof.addUsedClass(money);
116     rel.setStereotype("needs more");
117     rel.setLabel("has much");
118 
119     student.addSuperClass(person);
120 
121     // Add operations and class attributes aka fields
122     person.addOperation("setAddress", "void", "Public", new String[] { "Address" }, new String[] { "addr" });
123     person.addClassAttribute("age", "int");
124     person.addClassAttribute("name", "String");
125 
126     period.addClassAttribute("from", "Date");
127     period.addClassAttribute("to", "Date");
128 
129     /* Add association "teach": associations contain a "role list" which has exactly
130      * two members. The Role objects contain the information about aggregation, the
131      * cardinality, etc.
132      */
133     Association assoc = factory.createAssociation("teach", prof, student);
134     cat.addToModel(assoc);
135 
136     Role first  = assoc.getFirstRole();
137     Role second = assoc.getSecondRole();
138 
139     first.setRoleName("Talker");
140     first.isNavigable(false); // Only goes from professor to student
141     second.setRoleName("Listener");
142     second.setCardinality("1..n");
143     first.setConstraints("None");
144 
145     /* Note that the kind of aggregation depends on the containment of the
146      * other role. I.e., first (Professor) has an aggregation association of
147      * second (Student), but the containment kind has to be set on the latter.
148      */
149     first.setIsAggregate(true);
150     second.setContainment("By reference");
151 
152     // Add association between university and department
153     Association assoc2 = factory.createAssociation(uni, department);
154     cat.addToModel(assoc2);
155 
156     first  = assoc2.getFirstRole();
157     second = assoc2.getSecondRole();
158     first.setCardinality("1");
159     second.setCardinality("1..n");
160     first.setIsAggregate(true);
161     second.setContainment("By value");
162 
163     // Add association between professor and department
164     Association assoc3 = factory.createAssociation(prof, department);
165     cat.addToModel(assoc3);
166     assoc3.setAssociationClass(period.getQualifiedName());
167 
168     /* Set up views in class diagram, basically this reflects just the
169      * structures defined in the data model.Everything related to an
170      * entity such as stereotypes should be created automagically via
171      * the factory.  It's important to add classes first, since
172      * relationships, etc., rely on them.
173      * Views also have "tags", i.e., a local numbering scheme they be
174      * refered with, the tags, i.e. indexes are set automatically when
175      * the views are added.
176      */
177     ClassDiagram diag = cat.getFirstClassDiagram();
178 
179     ClassView studentview, profview; // We'll need these later on
180 
181     diag.addToView(factory.createClassView(person));
182     diag.addToView(profview = factory.createClassView(prof));
183     diag.addToView(studentview = factory.createClassView(student));
184     diag.addToView(factory.createClassView(mentor));
185     diag.addToView(factory.createClassView(address));
186     diag.addToView(factory.createClassView(money));
187     diag.addToView(factory.createClassView(uni));
188     diag.addToView(factory.createClassView(department));
189     diag.addToView(factory.createClassView(period));
190 
191     diag.addToView(factory.createAssociationView(assoc));
192     diag.addToView(factory.createAssociationView(assoc2));
193     diag.addToView(factory.createAssociationView(assoc3));
194 
195     /* Create note and attach it to student and profs, notes are not
196      * contained in the data model, just in the view.
197      */
198     NoteView noteview;
199     diag.addToView(noteview = factory.createNoteView("This is\nsome note"));
200     diag.addAttachView(noteview, profview); // Convenience method
201     diag.addAttachView(noteview, studentview);
202 
203     /** Ok, now add the inheritance views for the classes, i.e., some arrows.
204      */
205     for(Iterator<?> i = factory.createInheritViews(prof).iterator(); i.hasNext(); ) {
206       InheritView view = (InheritView)i.next();
207 
208       diag.addToView(view);
209     }
210 
211     for(Iterator<?> i = factory.createInheritViews(student).iterator(); i.hasNext(); ) {
212       InheritView view = (InheritView)i.next();
213 
214       diag.addToView(view);
215     }
216 
217     /** Do the same for the other kinds of associations.
218      */
219     for(Iterator<?> i = factory.createRealizeViews(prof).iterator(); i.hasNext(); ) {
220       RealizeView view = (RealizeView)i.next();
221 
222       diag.addToView(view);
223     }
224 
225     for(Iterator<?> i = factory.createUsesViews(prof).iterator(); i.hasNext(); ) {
226       UsesView view = (UsesView)i.next();
227 
228       diag.addToView(view);
229     }
230 
231     ///////////////// Create data and view for use case diagram /////////////
232 
233     UseCase lecture = factory.createUseCase("Lecture");
234     UseCase exam    = factory.createUseCase("Examination");
235     UseCase exam2   = factory.createUseCase("Hard examination");
236     UseCaseCategory cat2 = model.getUseCaseCategory();
237 
238     cat2.addToModel(lecture);
239     cat2.addToModel(exam);
240     cat2.addToModel(exam2);
241     exam2.addSuperUseCase(exam);
242 
243     lecture.setStereotype("obsolete");
244 
245     assoc = factory.createAssociation("depends on", exam, lecture);
246     cat2.addToModel(assoc);
247     assoc.getFirstRole().isNavigable(false);
248 
249     assoc2 = factory.createAssociation(prof, lecture);
250     cat2.addToModel(assoc2);
251     assoc2.getFirstRole().isNavigable(false);
252 
253     assoc3 = factory.createAssociation(prof, exam);
254     cat2.addToModel(assoc3);
255     assoc3.getFirstRole().isNavigable(false);
256 
257     UseCaseDiagram diag2 = cat2.getFirstUseCaseDiagram();
258 
259     diag2.addToView(factory.createUseCaseView(lecture));
260     diag2.addToView(factory.createUseCaseView(exam));
261     diag2.addToView(factory.createUseCaseView(exam2));
262     diag2.addToView(factory.createClassView(prof));
263 
264     for(Iterator<?> i = factory.createInheritViews(exam2).iterator(); i.hasNext(); ) {
265       InheritView view = (InheritView)i.next();
266 
267       diag2.addToView(view);
268     }
269 
270     diag2.addToView(factory.createAssociationView(assoc));
271     diag2.addToView(factory.createAssociationView(assoc2));
272     diag2.addToView(factory.createAssociationView(assoc3));
273 
274     // We're done, print the model.
275     ByteArrayOutputStream baos = new ByteArrayOutputStream();
276     PrintStream ps = new PrintStream(baos);
277     model.accept(new PrintVisitor(ps));
278     String petalContent=baos.toString();
279     assertTrue(petalContent.contains("Rose 4.5.8163.3"));
280     assertTrue(petalContent.length()>41000);
281     // debug=true;
282     if (debug)
283     	System.out.println(petalContent.length());
284   }
